如下的临时路径在 reload 时会不起作用
#+BEGIN_SRC python
import os
import sys


try:
    path = os.path.split(os.path.realpath(__file__))[0]
except NameError:
    path = os.getcwd() or os.getenv('PWD')
root = os.path.dirname(path)
sys.path.insert(0, root)
#+END_SRC
